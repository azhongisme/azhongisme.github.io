深度优先搜索

2023/7/17 algorithm

深度优先搜索所使用的策略就像其名字所隐含的：只要可能，就尽量深入
在搜索算法中，DFS(深度优先搜索)常常指利用递归函数方便地实现暴力枚举的算法，与图论中的DFS算法
有一定相似之处，但并不完全相同。深度优先搜索一般使用栈来实现。

八皇后问题

#include<bits/stdc++.h>
using namespace std;
int qipan[15][15];
int n;
bool st1[100],st2[100],st3[100];//1列，2左对角线，3右对角线
int ans = 0;
int a[100];
void dfs(int u){
    if(u==n){
        ans++;
        if(ans<=3){
            for(int i = 0;i<n;i++){
                cout<<a[i]<<" ";
            }
            cout<<"\n";
        }
    }
    for(int i = 1;i<=n;i++){
        //由于对角线是斜线，因此可以看成y = x + b，此处的+n是为了下标保证为正值
        if(!st1[i]&&!st2[i-u+n]&&!st3[i+u]){
            a[u] = i;
            st1[i] = st2[i-u+n] = st3[i+u] = 1;
            dfs(u+1);
            st1[i] = st2[i-u+n] = st3[i+u] = 0;
        }
    }
}
int main(){
    cin>>n;
    dfs(0);
    cout<<ans;
    system("pause");
}

[NOIP2000 提高组] 单词接龙

#include<bits/stdc++.h>
using namespace std;
const int maxn = 50;
int n;
int ans = 0;
string word[maxn];
string beginn;
//used数组判断该单词使用次数，小于2可以使用
int used[maxn];
//判断b是否能接到a的后面，k表示重叠部分的长度
bool check(string a,string b,int k){
    int len = a.length();
    for(int i = 0;i<k;i++){
        if(a[len-k+i]!=b[i]) return false;
    }
    return true;
}
//接龙
void add(string &a,string b,int k){
    for(int i = k;i<b.length();i++) a+=b[i];
}
void dfs(string now){
    int x = now.length();
    ans = max(x,ans);
    for(int i = 0;i<n;i++){
        if(used[i]>=2){
            continue;
        }
        int len = word[i].length();
        for(int j = 1;j<=len;j++){
            if(check(now,word[i],j)){
                used[i]++;
                string tmp = now;
                add(tmp,word[i],j);
                dfs(tmp);
                used[i]--;
            }
        }
    }
}
int main(){
    cin>>n;
    for(int i = 0;i<n;i++) cin>>word[i];
    cin>>beginn;
    dfs(beginn);
    cout<<ans;
    system("pause");
}

#include<bits/stdc++.h>
using namespace std;
const int N = 1010;
bool flag = false;
void dfs(int n,int m){
    if(n==m){
        flag = true;
        return;
    }
    if(n<m) return;
    if(n%3) return;
    dfs(n/3,m);
    dfs(n/3*2,m);
}
//简化版本的dfs
void dfs1(int n,int m){
    return n==m ? 1 : (n<m||n%3) ? 0 : (dfs(n/3)||dfs(n/3*2));

E. The Lakes
题目大意，一个n*m的矩阵，求不为0的连通块最大之和
分析：这道题用BFS更好一点，但这里是DFS,那就要DFS写了

#include <bits/stdc++.h>
using namespace std;
int dx[5]={0,0,0,1,-1};
int dy[5]={0,1,-1,0,0};
int n,m;
vector<int> ans(2002000);
vector<vector<int> > a(2010,vector<int>(2010));
//vis数组判断该点是否搜过
vector<vector<int> > vis(2010,vector<int>(2010));
//cnt表示第cnt个连通块
void dfs(int x, int y,int cnt)
{
    if(vis[x][y]||x<0||x>=n||y<0||y>=m||a[x][y]==0)
    {
        return;
    }
    vis[x][y]=1;
    ans[cnt]+=a[x][y];
    for(int i=1;i<=4;i++)
    {
        int x1=x+dx[i];
        int y1=y+dy[i];
        //是否越界
        if(x1<0||x1>=n||y1<0||y1>=m)continue;
        dfs(x1,y1,cnt);
    }
    return;
}
void solve() {
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            vis[i][j]=0;
            a[i][j]=0;
        }
    }
    int cnt=-1;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            cin>>a[i][j];
        }
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            //该点不为0，且未被用过
            if(a[i][j]&&!vis[i][j])
            {
                cnt++;
                dfs(i,j,cnt);
            }
        }
    }
    int max1=0;
    for(int i=0;i<=cnt;i++)
    {
        max1=max(max1,ans[i]);
        ans[i]=0;
    }
    cout<<max1<<endl;
}

完整代码如下

#include<bits/stdc++.h>
using namespace std;
const int maxn=10;
const double PI=3.1415926535;
bool s[maxn];
double x[maxn],y[maxn],r[maxn],xa,ya,xb,yb,ansmax;
int n;
void dfs(int k,double sum){
    if(k>n){
        ansmax = max(ansmax,sum);
        return;
    }
    for(int i = 1;i<=n;i++){
        if(!s[i]){
            s[i] = 1;
            double _ = min(abs(x[i]-xa),abs(x[i]-xb));
            double __ = min(abs(y[i]-yb),abs(y[i]-ya));
            r[i] = min(_,__);
            for(int j = 1;j<=n;j++){
                if(s[j]&&i!=j){
                    double d = (x[i]-x[j])*(x[i]-x[j]) + (y[i]-y[j])*(y[i]-y[j]);
                    d = sqrt(d);
                    r[i] = min(r[i],max(d-r[j],0.0));
                }
            }
            dfs(k+1,sum+r[i]*r[i]*PI);
            s[i] = 0;
            r[i] = 0;
        }
    }
}
int main(){
    double ss;
    cin>>n;
    cin>>xa>>ya>>xb>>yb;
    ss = abs(xa-xb)*abs(ya-yb);
    for(int i = 1;i<=n;i++) cin>>x[i]>>y[i];
    dfs(1,0);
    cout<<(int)(ss-ansmax+0.5);
    system("pause");
}
